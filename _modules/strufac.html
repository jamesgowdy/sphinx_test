    '''Parse a PDB file for info needed to calculate structure factors
    The PDB is parsed according to the following `format described here`_.
    As well as the atom and hetatm records the parser will extract:
    
    * The symmetry operators from the mandatory remark 290 record
    * The unit cell parameters from the cryst1 record
    * The deorthoganilsation matrix from the scale record.

    A RuntimeError is raised if these are not available. We use the scale matrix
    to transform the cartesian world coordinates and symmetry operators to the
    fractional crystallographic basis.

    :param str pdbfile: path to a PDB file
    :return: instance of :class:`ParsePDB <ParsePDB>` with the following attributes
    :ivar xyz_w: world coordinates for the assymetric unit as column vectors 
    :ivar xyz_c: fractional coordinates for the assymetric unit as column vectors
    :ivar n: occupancies
    :ivar Biso: isotropic B-factors
    :ivar e: element symbols
    :ivar S: the scale matrix for deorthoganlisation
    :ivar abc: unit cell lengths (in Angstroms)
    :ivar angles: unit cell angles (in radians) 
    :ivar Rw: list of symops rotation matricies for cartesian basis 
    :ivar tw: list of symops translation vectors for catesian basis
    :ivar Rc: list of symops rotation matricies for crystallographic basis
    :ivar tc: list of symops operator translation vectors for crystallographic basis 

    .. _format described here: http://deposit.rcsb.org/adit/docs/pdb_atom_format.html
    '''

    def __init__(self, pdbfile):

        from collections import defaultdict
        
        # initalise attributes
        for var in ['xw', 'yw', 'zw', 'n', 'Biso', 'e', 'S', 'symops', 'Rw', 'tw']: 
            vars(self)[var] = []
        Rw, Tw = defaultdict(list), defaultdict(list)

        for line in pdbfile.split('\n'):

            if line.startswith('ATOM') or line.startswith('HETATM'):
                # atomic coordinates etc.
                self.xw.append(float(line[30:38])) 
                self.yw.append(float(line[38:46])) 
                self.zw.append(float(line[46:54])) 
                self.n.append(float(line[54:60])) 
                self.B.append(float(line[60:66])) 
                self.e.append(str.title(str.strip(line[76:78]))) 
                
            elif line.startswith('CRYST1'):
                # unit cell parameters
                self.abc = map(float, [line[6:15], line[15:24], line[24:33]]) 
                angles = [line[33:40], line[40:47], line[47:54]]
                self.angles = [np.radians(float(ang)) for ang in angles] 
                
            elif line.startswith('SCALE'):
                # deorthoganlisation 'SCALE' matrix
                #self.offset.append(float(line[45:55])) 
                assert float(line[45:55]) == 0., 'Non-zero offset?' 
                row = map(float, [line[10:20], line[20:30], line[30:40]])
                self.S.append(row)
                
            elif line.startswith('REMARK 290   SMTRY'):
                # rotation and translation matricies for cartesian/world basis
                line = line.split()
                Rw[int(line[3])].append(map(float, line[4:7]))
                Tw[int(line[3])].append(float(line[7]))
                
            elif line.startswith('REMARK 290'):
                # symbolic symmetry operator for crystallographic basis
                line = line.split()
                if len(line) == 4 and line[2].isdigit():
                    self.symops.append(line[3])
        
        # make sure file has everything we expect
        if not len(Rw): raise RuntimeError('No symmetry operators in PDB file')
        if not len(self.S): raise RuntimeError('No SCALE matrix in PDB file')
        if not hasattr(self, 'abc'): raise RuntimeError('No CRYST1 record in PDB file')

        # convert lists to numpy arrays
        for var in ['xw', 'yw', 'zw', 'n', 'Biso', 'e', 'S']: 
            vars(self)[var] = np.array(vars(self)[var])
        for key in Rw:
            self.Rw.append(np.array(Rw[key]))
            self.tw.append(np.array(tw[key]))

        # transform to coords and symops to crystallographic fractional basis 
        self.xyz_w = np.row_stack((self.x, self.y, self.z))

        SI = np.linalg.inv(self.S)
        self.Rc = [np.dot(self.S, R, SI) for R in self.Rw]
        self.tc = [np.dot(self.S, tw) for tw in self.tw]



[docs]class UnitCell:
    r'''Generate all atoms in the unit cell from the assymetric unit repesented by
    the PDB file using the symmetry operators. Special positions are removed.

    :param pdb: instance of :class:`ParsePDB <ParsePDB>`
    :return: instance of :class:`UnitCell<unitCell>` with the following attributes
    :ivar xyz_w: world coordinates for the unit cell 
    :ivar xyz_c: x fractional coordinates for the unit cell 
    :ivar n: occupancies 
    :ivar S: the scale matrix for deorthoganlisation
    :ivar Biso: isotropic B-factors
    :ivar e: element symbols
    '''
    def __init__(self, pdb):

        self.S = pdb.S
        ASUc, UCc = pdb.xyz_c, pdb.xyz_c.copy()
        ASUw, UCw = pdb.xyz_w, pdb.xyz_w.copy()

        tol = 1e-4
        for Rc, tc, Rw, tw in zip(pdb.Rc, pdb.tc, pdb.Rw, pdb.tw):
            # ignore identity operator
            if max(abs(np.eye(3)-Rc)) < tol and max(abs(tc)) < tol: 
                continue
            
            UCw = np.column_append(UCw, np.dot(Rw, ASUw))
            UCc = np.column_append(UCc, np.dot(Rc, ASUc))

        # locate special positions
        n = len(pdb.Rc)
        self.e = np.repeat(pdb.e, n)
        unique, special = locate_specials(UCw, self.e)

        # remove special positions
        self.e = self.e[unique]
        self.Biso = np.repeat(pdb.Biso, n)[unique]
        self.n = np.repeat(pdb.n, n)[unique]
        self.xyz_c = UCc[unique]
        self.xyz_w = UCw[unique]



[docs]def locate_specials(xyz, elems, neglogtol=4):
    '''Find unique atom coordinates as columns in a matrix in order to 
    exclude special position atoms on symmetry axes.

    :param np.ndarray xyz: a 2D array with coordinates stored as column vectors
    :param list elems: a list of elements symbols
    :kwarg int neglogtol: the negative logarithm of the tolerance
    :returns: indcies of the general and special positions ``(general, special)``
    :rtype: (1D `np.ndarray`, 1D `np.ndarray`)
    '''
    three, n = xyz.shape
    if three != 3: raise ValueError('xyz array not correct shape') 
    if n != len(elems): raise ValueError('Arrays not the same length')

    # round to ensure will eliminate if within tolerance
    xyz = np.around(xyz, decimals=neglogtol)

    # set up structured array
    arr = np.zeros(n, dtype=[('elems','a10'),('xyz','f4',(3,))])
    arr['elems'], arr['xyz'] = np.array(elems), xyz

    unique, unique_idx = np.unique(arr, return_index=1)
    special = np.ones(n)
    special[unique_idx] = 0
    return unique_idx, np.where(special)



[docs]def frac_symops(S, Rw, tw):
    r'''Transform the symmetry opertators for use with coordinates in the 
    fractional crystallographic basis

    :param np.ndarray S: scale matrix from :class:`ParsePDB <ParsePDB>`
    :param np.ndarray Rw: a cartesian basis rotation matrix
    :param np.ndarray tw: a catesian basis translation vector
    :return: transformed rotation matrix and translation vector ``(Rc, tc)``
    :rtype: a tuple of (2D `np.ndarray`, 1D `np.ndarray`)
    '''
    SI = np.linalg.inv(S)
    Rc = np.dot(S, Rw, SI)
    tc = np.dot(S, tw)
    return Rc, tc

    r'''Transform cartesian world coordinates into the fractional 
    crystallographic basis.

    :param np.ndarray S: scale matrix from :class:`ParsePDB <ParsePDB>`
    :param xyz: coordinates with the coefficents expressed in the certesian basis
        stored as column vectors 
    :type xyz: 1D np.ndarray or 2D np.ndarray 
    :return: transformed fractional coordinates stored as column vectors
    :rtype: 2D or 1D np.ndarray
    '''        
    return np.dot(S, xyz)



[docs]def hklmax(S, dmin):
    '''Calculate a maximum bound on hkl corresponding to a minimum resolution
    i.e. the reciprocal lengths of the vectors colinear with the unit cell 
    basis vectors scaled by *dmin*.

    :param np.ndarray S: scale matrix from :class:`ParsePDB <ParsePDB>`
    :param float dmin: minimum interplanar distance of interest
    :return: Miller indicies as a tuple: ``(hmax, kmax, lmax)``
    :rtype: (float, float, float)
    '''
    hmax = 1./(dmin*np.linalg.norm(S[0])) # calc_d(ST,[dmin,0,0])
    kmax = 1./(dmin*np.linalg.norm(S[1]))
    lmax = 1./(dmin*np.linalg.norm(S[2]))
    return hmax, kmax, lmax



[docs]def resolution(S, hkl):
    r'''Calculate the resolution (aka the interplanar distance) corresponding 
    to a given reflection
    
    :param np.ndarray S: the scale matrix from :class:`ParsePDB <ParsePDB>`
    :param hkl: Miller index vector or matrix with each triple stored as columns
    :type hkl: 1D np.ndarray or 2D np.ndarray
    :return: the interplanar distance(s)
    :rtype: floar or `np.ndarray` 
    '''
    ST = np.transpose(S)
    norm = lambda A: np.sum(A*A, axis=0)**0.5
    dstar = norm(np.dot(ST, hkl))
    return 1./dstar



[docs]def stol(d):
    r'''Convert an interplanar distance to :math:`\frac{\sin(\theta)}{\lambda}`
    using Bragg's law.

    :param d: the interplanar distance
    :type d: float or 1D np.ndarray
    :return: :math:`\frac{\sin(\theta)}{\lambda}`
    :rtype: `float` or 1D `np.ndarray`
    '''
    return 1./(2*d)

    '''Return a matrix of hkl indicies that fall within a resolution range 
    [*dmin*, *dmax*] as a 2D matrix with dimensions *(3, N)* also returns the 
    corresponding interplanar distances, *d*, as a vector of length *N*

    :param S: the scale matrix from :class:`ParsePDB <ParsePDB>`
    :param float dmin: minimum resolution (interplanar distance)
    :kwarg float dmax: optional maximum resolution (interplanar distance) 
    :return: hkl indicies and interplanar distance vector:``(hkls, ds)``
    :rtype: (2D `np.ndarray`, 1D `np.ndarray`)
    '''
    hkls = np.meshgrid(*map(range, hklmax(S, dmin)))
    hkls = np.row_stack(map(np.ndarray.flatten, hkls))[:,1:]
    ds = resolution(S, hkls)
    mask = dmin <= ds <= dmask
    return hkls[:, mask], ds[mask]

[docs]def calc_f0(e, stol):
    r'''Calculate an atomic scattering factor at a given resolution
    of :math:`\sin(\theta)/\lambda`.

    :param str e: atomic symbol
    :param float stol2: the resolution expressed as :math:`\sin(\theta)/\lambda`
    :return: the atomic scattering factor at the specified resolution
    :rttype: `float`
    '''
    a,b,c = CMA[e], CMB[e], CMC[e]
    stol2 = stol*stol
    f0 = sum(a[i]*exp(-b[i]*stol2) for i in range(4))+c
    return f0



[docs]def precalc_fB(uc, hkl):
    '''Pre-calculate scattering factors with isotropic B-factor and occupancy 
    contributions for all atoms in the unit cell at a specified resolution 
    dictated by the Miller index.

    :param uc: instance of :class:`UnitCell <UnitCell>`
    :param numpy.ndarray hkl: a Miller index triple
    :return: scattering factors (``fBs``)
    :rtype: 1D `numpy.ndarray`
    '''
    f0, fB = {}, []
    stol = stol(resolution(uc.S, hkl))
    stol2 = stol*stol

    for e in set(uc.e):
        f0[e] = calc_scatfact(e, stol)
    for e, Biso, n in zip(uc.e, uc.Biso, uc.n):
        debye_waller = np.exp(-Biso*stol2)
        fB.append(f0[e]*n*debye_waller)

    return np.array(fB)



[docs]def directsum_F(uc, fBs, hkl):
    '''Calculate a structure factor by direct summation

    :param uc: instance of :class:`UnitCell <UnitCell>`
    :param np.ndarray fBs: precalculated scattering factors with B-factor contributions
    :param numpy.ndarray hkl: a Miller index triple
    :return: a complex structure factor
    :rtype: `complex` scalar
    '''
    F = 0
    pi2j = np.pi*2j
    for xyz, fB in zip(uc.xyz_c, fBs):
        hx = np.sum(hkl*xyz)
        F += fB*np.exp(pi2j*hx)
    return F
