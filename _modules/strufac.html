[docs]class ParsePDB:
    '''Parse a PDB file for data needed to calculate structure factors
	
	The PDB is parsed according to the following format described [here]
	(http://deposit.rcsb.org/adit/docs/pdb_atom_format.html), the symmetry
	operators are extracted from the mandatory REMARK 290 record.

    :param: path to a PDB file
	:type: str

	:ivar x: x-coordinate (cartesian world basis)
	:ivar y: y-coordinate (cartesian world basis)
	:ivar z: z-coordinate (cartesian world basis)
	:ivar n: atomic occupancies
	:ivar B: isotropic B-factors
	:ivar e: element symbol
	:ivar S: deorthoganlisation matrix
	:ivar abc: unit cell lengths (in Angstroms)
	:ivar angles: unit cell angles (in radians) 
	:ivar Rw: symmetry operator rotation matricies for cartesian basis 
	:ivar Tw: symmetry operator translation vector for catesian basis
    '''

    def __init__(self, pdbfile):
        
        # initalise attributes
        for var in ['x','y','z','n','Biso','e','S','offset','symops']: 
            vars(self)[var] = []
        self.Rw, self.Tw = defaultdict(list), defaultdict(list)
        
        for line in pdbfile.split('\n'):

            if line.startswith('ATOM') or line.startswith('HETATM'):
            	# atomic coordinates etc.
                self.x.append(float(line[30:38])) 
                self.y.append(float(line[38:46])) 
                self.z.append(float(line[46:54])) 
                self.n.append(float(line[54:60])) 
                self.B.append(float(line[60:66])) 
                self.e.append(str.title(str.strip(line[76:78]))) 
                
            elif line.startswith('CRYST1'):
            	# unit cell parameters
                self.abc = map(float, [line[6:15], line[15:24], line[24:33]]) 
                angles = [line[33:40], line[40:47], line[47:54]]
                self.angles = [radians(float(ang)) for ang in angles] 
                
            elif line.startswith('SCALE'):
                # deorthoganlisation 'SCALE' matrix 
                row = map(float, [line[10:20], line[20:30], line[30:40]])
                self.S.append(row)
                self.offset.append(float(line[45:55])) 
                assert float(line[45:55]) == 0., 'Non-zero offset?'
                
            elif line.startswith('REMARK 290   SMTRY'):
                # rotation and translation matricies for cartesian/world basis
                line = line.split()
                self.Rw[int(line[3])].append(map(float, line[4:7]))
                self.Tw[int(line[3])].append(float(line[7]))
                
            elif line.startswith('REMARK 290'):
                # symbolic symetry operator for crystallographic basis
                line = line.split()
                if len(line) == 4 and line[2].isdigit():
                    self.symops.append(line[3])
                    
        assert len(self.Rw), 'No symmetry operators'
